### 4) 어떻게 고쳤나요?
```c
// 고친 코드를 여기에 적어주세요
```#include <stdio.h>

#define MAX 100

char names[MAX][20];
int count = 0;

int main() {
    int menu;

    while (1) {
        printf("\n1. 출석 체크\n2. 명단 보기\n3. 종료\n선택: ");
        scanf("%d", &menu);

        if (menu == 1) {
            if (count >= MAX) {
                printf("⚠️ 출석 인원이 이미 100명입니다! 더 이상 입력할 수 없습니다.\n");
                continue;
            }

            printf("이름: ");
            scanf("%19s", names[count]); // 버퍼 오버플로 방지
            printf("출석 완료!\n");
            count++;
        }
        else if (menu == 2) {
            if (count == 0) {
                printf("아직 등록된 사람이 없습니다.\n");
            } else {
                printf("\n--- 출석 명단 (%d명) ---\n", count);
                for (int i = 0; i < count; i++)
                    printf("%d. %s\n", i + 1, names[i]);
            }
        }
        else if (menu == 3) {
            printf("종료합니다.\n");
            break;
        }
        else {
            printf("잘못 입력!\n");
        }
    }

    return 0;
}

## 💭 디버깅하면서 배운 점

이번 프로젝트에서 오류를 고치면서 새롭게 알게 된 점이나 느낀 점을 자유롭게 적어보세요.

```이번 프로그램을 디버깅하면서 단순히 코드가 돌아가는 것뿐 아니라, 프로그램이 어떤 상황에서도 정상적으로 작동하도록 만드는 것이 얼마나 중요한지 깨달았다. 처음에는 단순히 이름을 입력하고 출력하는 기능만 구현하면 된다고 생각했지만, 실제로 테스트해보니 입력 개수가 제한보다 많아지는 경우나 잘못된 값을 입력했을 때 프로그램이 예기치 않게 종료되거나 오류가 발생하는 문제들을 발견할 수 있었다.

특히 이번 작업을 통해 에러는 단순한 문제나 방해 요소가 아니라, 프로그램을 더 안정적이고 완성도 있게 만들어주는 힌트라는 것을 느꼈다. 디버깅 과정에서는 단순히 오류를 고치는 것뿐 아니라, 사용자가 어떻게 사용할지 미리 생각하고 조건을 추가하거나 예외 처리를 하는 과정이 중요하다는 점도 알게 되었다.

또한 작은 수정 하나에도 프로그램의 전체 흐름과 구조를 고려해야 한다는 점에서 논리적 사고와 꼼꼼함이 필요하다는 것도 깨닫게 되었다. 이번 디버깅 경험은 코드 실행 자체보다, 프로그램이 안전하고 예상 가능한 방식으로 작동하도록 만드는 과정의 중요성을 배울 수 있었던 의미 있는 경험이었다.
(예시)
- 세미콜론(;)을 빼먹으면 컴파일 오류가 난다는 것을 배웠어요.
- scanf를 쓸 때 &를 꼭 써야 한다는 걸 알았어요.
- 에러 메시지를 잘 읽으면 문제를 더 쉽게 찾을 수 있었어요.
```
